/**
 * @module queries/users
 * @description Consultas SQL para la tabla users.
 */

const queries = {
    /**
   * Obtener un usuario por email.
   * @type {string}
   */
  getUser: `
    SELECT * FROM 
        users
    WHERE 
        email = $1;
    `,
    /**
   * Obtener informaci√≥n del perfil de un usuario.
   * @type {string}
   */
  getInfoProfile: `
    SELECT name, email, role
    FROM users
    WHERE id_user = $1;
  `,
    /**
   * Crear un nuevo usuario.
   * @type {string}
   */
  createUser: `
    INSERT INTO users (name, email, password, role)
    VALUES ($1, $2, $3, $4)
    RETURNING id_user;
    `,
  /**
   * Eliminar un usuario.
   * @type {string}
   */
  deleteUser: `
    DELETE FROM users
    WHERE id_user = $1
    RETURNING id_user, name, email, role;
        `,
};
module.exports = queries;
